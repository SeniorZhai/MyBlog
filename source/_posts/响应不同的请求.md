```javascript
node
var url = require('url')
var requestURL = 'http://example.com:1234/pathname?query=string# hash'
// 获取主机名
url.parse(requestURL).hostname
// 获取路径端口
url.parse(requestURL).port
// 获取路径名
url.parse(requestURL).pathname
```
